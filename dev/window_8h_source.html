235  // We have been blessed with so much reflection that it's actually seems evil now.
236  template <typename T>
237  constexpr ws::uid add_event_notifier(T &evt) {
238  const std::lock_guard guard_(ws_client_lock);
239 
240  if constexpr (std::is_same_v<T, epoc::event_mod_notifier_user>) {
241  mod_notifies.emplace(std::move(evt));
242  return static_cast<ws::uid>(mod_notifies.size());
243  } else if constexpr (std::is_same_v<T, epoc::event_screen_change_user>) {
244  screen_changes.emplace(std::move(evt));
245  return static_cast<ws::uid>(screen_changes.size());
246  } else if constexpr (std::is_same_v<T, epoc::event_error_msg_user>) {
247  error_notifies.emplace(std::move(evt));
248  return static_cast<ws::uid>(error_notifies.size());
249  } else if constexpr (std::is_same_v<T, epoc::event_capture_key_notifier>) {
250  return add_capture_key_notifier_to_server(evt);
251  } else {
252  throw std::runtime_error("Unsupported event notifier type!");
253  }
254  }
255  };
