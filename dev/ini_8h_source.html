21 /* This ini parser is made to deal with many INI types across Symbian OS
22  *
23  * There includes wsini, cenrep ini, etc.. These INI are advances from the original INI
24  * format and each server has it own parser.
25  *
26  * Here i group them all, and keep struct as lightweight as possible.
27  *
28  * Parsing support
29  * Two types of prop (pair):
30  *
31  * INT A B (Prop seperate with space)
32  * A=5, 7, 9 (Prop equals + seperate with comma)
33  *
34  * These can be recursive, but don't test it to its limit:
35  *
36  * INT A = 5, 7, 9 B
37  *
38  * Will be as:
39  * PROP KEY = INT, VALUES = { {KEY = A, VALUE = {5, 7, 9} } , B }
40 */

106  template <typename T>
107  std::enable_if_t<std::is_integral_v<T>, T> get_as_native() {
108  std::string vc = value;
109 
110  if (vc.length() > 2 && vc[0] == '0' && (vc[1] == 'x' || vc[1] == 'X')) {
111  vc = vc.substr(2, vc.length() - 2);
112  return static_cast<T>(std::strtoll(vc.c_str(), nullptr, 16));
113  }
114 
115  return static_cast<T>(std::atoll(vc.c_str()));
116  }
117

119  std::enable_if_t<std::is_floating_point_v<T>, T> get_as_native() {
120  return static_cast<T>(std::stod(value));
