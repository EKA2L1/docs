149  path_iterator operator++(int dummy) {
150  if (crr_pos > path.length()) {
151  throw std::runtime_error("Iterator is invalid");
152  return *this;
153  }
154 
155  if (crr_pos < path.length()) {
156  comp = "";
157  } else {
158  crr_pos++;
159  return *this;
160  }
161 
162  while (crr_pos < path.length() && !is_separator(path[crr_pos])) {
163  comp += path[crr_pos];
164  crr_pos += 1;
165  }
166 
167  while (crr_pos < path.length() && is_separator(path[crr_pos])) {
168  crr_pos += 1;
169  }
170 
171  return *this;
172  }
