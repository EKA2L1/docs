82  template <typename T>
83  constexpr kernel::object_type get_object_type() {
84  if constexpr (std::is_same_v<T, kernel::process>) {
85  return kernel::object_type::process;
86  } else if constexpr (std::is_same_v<T, kernel::thread>) {
87  return kernel::object_type::thread;
88  } else if constexpr (std::is_same_v<T, kernel::chunk>) {
89  return kernel::object_type::chunk;
90  } else if constexpr (std::is_same_v<T, kernel::library>) {
91  return kernel::object_type::library;
92  } else if constexpr (std::is_same_v<T, kernel::mutex>) {
93  return kernel::object_type::mutex;
94  } else if constexpr (std::is_same_v<T, kernel::semaphore>) {
95  return kernel::object_type::sema;
96  } else if constexpr (std::is_same_v<T, kernel::timer>) {
97  return kernel::object_type::timer;
98  } else if constexpr (std::is_same_v<T, service::property>) {
99  return kernel::object_type::prop;
100  } else if constexpr (std::is_same_v<T, service::server>) {
101  return kernel::object_type::server;
102  } else if constexpr (std::is_same_v<T, service::session>) {
103  return kernel::object_type::session;
104  } else if constexpr (std::is_same_v<T, kernel::codeseg>) {
105  return kernel::object_type::codeseg;
106  } else if constexpr (std::is_same_v<T, kernel::change_notifier>) {
107  return kernel::object_type::change_notifier;
108  } else {
109  throw std::runtime_error("Unknown kernel object type. Make sure to add new type here");
110  return kernel::object_type::unk;
111  }
112  }

296  template <typename T>
297  constexpr std::shared_ptr<T> get_by_name_and_type(const std::string &name, const kernel::object_type obj_type) {
298  switch (obj_type) {
299 #define OBJECT_SEARCH(obj_type, obj_map) \
300  case kernel::object_type::obj_type: { \
301  auto res = std::find_if(obj_map.begin(), obj_map.end(), [&](const auto &rhs) { \
302  return name == rhs->name(); \
303  }); \
304  if (res == obj_map.end()) \
305  return nullptr; \
306  return std::reinterpret_pointer_cast<T>(*res); \
307  }
308 
309  OBJECT_SEARCH(mutex, mutexes)
310  OBJECT_SEARCH(sema, semas)
311  OBJECT_SEARCH(chunk, chunks)
312  OBJECT_SEARCH(thread, threads)
313  OBJECT_SEARCH(process, processes)
314  OBJECT_SEARCH(change_notifier, change_notifiers)
315  OBJECT_SEARCH(library, libraries)
316  OBJECT_SEARCH(codeseg, codesegs)
317  OBJECT_SEARCH(server, servers)
318  OBJECT_SEARCH(prop, props)
319  OBJECT_SEARCH(session, sessions)
320  OBJECT_SEARCH(timer, timers)
321 
322 #undef OBJECT_SEARCH
323 
324  default:
325  break;
326  }

343  template <typename T>
344  constexpr std::shared_ptr<T> get_by_id(const kernel::uid uid) {
345  SYNCHRONIZE_ACCESS;
346 
347  constexpr kernel::object_type obj_type = get_object_type<T>();
348 
349  switch (obj_type) {
350  // It's gurantee that object are sorted by unique id, by just adding it
351  // Deletion will not affect that fact.
352 #define OBJECT_SEARCH(obj_type, obj_map) \
353  case kernel::object_type::obj_type: { \
354  auto res = std::lower_bound(obj_map.begin(), obj_map.end(), nullptr, [&](const auto &lhs, const auto &rhs) { \
355  return lhs->unique_id() < uid; \
356  }); \
357  if (res != obj_map.end()) { \
358  return std::reinterpret_pointer_cast<T>(*res); \
359  } \
360  return nullptr; \
361  }

388  constexpr std::shared_ptr<T> create(args... creation_arg) {
389  constexpr kernel::object_type obj_type = get_object_type<T>();
390  std::shared_ptr<T> obj;
391 
392  if constexpr (obj_type == kernel::object_type::server) {
393  obj = std::make_shared<T>(sys, creation_arg...);
394  } else {
395  obj = std::make_shared<T>(this, creation_arg...);
396  }
397 
398  const kernel::uid obj_uid = obj->unique_id();
399 
400  switch (obj_type) {
401  case kernel::object_type::thread: {
402  threads.push_back(std::move(std::reinterpret_pointer_cast<kernel::thread>(obj)));
