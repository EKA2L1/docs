39  void decompress_rle(common::ro_stream *source, common::wo_stream *dest) {
40  static_assert(BIT % 8 == 0, "This RLE decompress function don't support unaligned bit decompress!");
41 
42  while (source->valid() && dest->valid()) {
43  std::int8_t count8 = 0;
44  source->read(&count8, 1);
45 
46  std::int32_t count = count8;
47 
48  constexpr int BYTE_COUNT = static_cast<int>(BIT) / 8;
49 
50  if (count >= 0) {
51  count = common::min(count, static_cast<const std::int32_t>(dest->left() / BYTE_COUNT));
52  std::uint8_t comp[BYTE_COUNT];
53 
54  for (std::size_t i = 0; i < BYTE_COUNT; i++) {
55  source->read(&comp[i], 1);
56  }
57 
58  while (count >= 0) {
59  for (std::size_t i = 0; i < BYTE_COUNT; i++) {

64  }
65  } else {
