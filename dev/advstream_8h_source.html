33  /* \brief A simple but yet advance stream
34  *
35  * Given the stream a buffer pointer, and its size, you can peek, and read from this stream like normal.
36  */
37  template <typename T>
38  class advstream {
39  T *data;
40  uint64_t count;
41  uint64_t size;
42 
43  protected:
44  T *_get(uint64_t hm) {
45  if (eof()) {
46  return nullptr;
47  }
48 
49  uint64_t get_count = std::min(hm, length());
50  T &res = data[get_count];
51 
52  count += get_count;
53 
54  return &res;
55  }
56 
57  T *_peek(uint64_t hm) const {
58  if (eof()) {
59  return nullptr;
60  }
61 
62  return &data[count];
63  }
64 
65  public:
66  advstream()
67  : count(0)
68  , data(nullptr) {}
69 
70  advstream(T *dat, const uint64_t ssize)
71  : data(dat)
72  , size(ssize)
73  , count(0) {}
74 
75  bool eof() const {
76  return count >= size;
77  }
78 
79  uint64_t length() const {
80  return size - count;
81  }
82  };
