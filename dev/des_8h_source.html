97  std::basic_string<T> to_std_string(eka2l1::kernel::process *pr) {
98  const des_type dtype = get_descriptor_type();
99  assert((dtype >= buf_const) && (dtype <= ptr_to_buf));
100 
101  std::basic_string<T> data;
102  data.resize(get_length());
103 
104  T *data_pointer_guest = get_pointer(pr);
105  std::copy(data_pointer_guest, data_pointer_guest + data.length(), &data[0]);
106 
107  return data;
108  }

120  int assign(eka2l1::kernel::process *pr, const std::uint8_t *data,
121  const std::uint32_t size) {
122  std::uint8_t *des_buf = reinterpret_cast<std::uint8_t *>(get_pointer_raw(pr));
123  des_type dtype = get_descriptor_type();
124 
125  std::uint32_t real_len = size / sizeof(T);
126 
127  if (size > 0) {
128  if ((dtype == buf) || (dtype == ptr) || (dtype == ptr_to_buf)) {
129  if (real_len > get_max_length(pr)) {
130  return des_err_not_large_enough_to_hold;
131  }
132  }
133 
134  std::memcpy(des_buf, data, size);
135  }
136 
137  set_length(pr, real_len);
138 
139  return 0;
140  }
