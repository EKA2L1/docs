47  template <typename T>
48  std::optional<T> pop() {
49  if (data.size() < sizeof(T)) {
50  return std::optional<T>{};
51  }
52 
53  T ret = *(reinterpret_cast<T *>(&data[0]));
54  data.erase(0, sizeof(T));
55 
56  return ret;
57  }
58 
62  template <typename T>
63  void push(const T val) {
64  data.append(reinterpret_cast<const char *>(&val), sizeof(val));
65  }
66 
67  void push_string(const std::string val) {
68  std::uint32_t len = static_cast<std::uint32_t>(val.length());
69  push(len);
70  data.append(val.begin(), val.end());
71  }
72 
73  std::optional<std::string> pop_string() {
74  auto len = pop<std::uint32_t>();
75 
76  if (!len || *len > data.size()) {
77  return std::nullopt;
78  }
79 
80  std::string ret = data.substr(0, *len);
81  data.erase(0, *len);
82 
83  return ret;
84  }
