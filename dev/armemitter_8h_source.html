239  operand2(arm_reg base, shift_type stype, std::uint8_t shift) // For IMM shifted register
240  {
241  if (shift == 32)
242  shift = 0;
243  switch (stype) {
244  case ST_LSL:
245  LOG_ERROR_IF(shift < 32, "Invalid operand2: LSL %u", shift);
246  break;
247  case ST_LSR:
248  LOG_ERROR_IF(shift <= 32, "Invalid operand2: LSR %u", shift);
249  if (!shift)
250  stype = ST_LSL;
251  if (shift == 32)
252  shift = 0;
253  break;
254  case ST_ASR:
255  LOG_ERROR_IF(shift < 32, "Invalid operand2: ASR %u", shift);
256  if (!shift)
257  stype = ST_LSL;
258  if (shift == 32)
259  shift = 0;
260  break;
261  case ST_ROR:
262  LOG_ERROR_IF(shift < 32, "Invalid operand2: ROR %u", shift);
263  if (!shift)
264  stype = ST_LSL;
265  break;
266  case ST_RRX:
267  LOG_ERROR_IF(shift == 0, "Invalid operand2: RRX does not take an immediate shift amount");
268  stype = ST_ROR;
269  break;
270  }
271  index_or_shift = shift;
272  shift = stype;
273  value = base;
274  type = TYPE_IMMSREG;
275  }

340  std::uint32_t Imm12Mod() const {
341  // This is an IMM12 with the top four bits being rotation and the
342  // bottom eight being an IMM. This is for instructions that need to
343  // expand a 8bit IMM to a 32bit value and gives you some rotation as
344  // well.
345  // Each rotation rotates to the right by 2 bits
346  LOG_ERROR_IF((type == TYPE_IMM), "Imm12Mod not IMM");
347  return ((rotation & 0xF) << 8) | (value & 0xFF);
348  }

350  std::uint32_t Imm16() const {
351  LOG_ERROR_IF((type == TYPE_IMM), "Imm16 not IMM");
352  return ((value & 0xF000) << 4) | (value & 0x0FFF);
353  }

362  return (((value >> 16) & 0xF000) << 4) | ((value >> 16) & 0x0FFF);
363  }
