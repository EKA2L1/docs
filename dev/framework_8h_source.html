41  public:
42  template <typename T>
43  T *get(const service::uid id) {
44  auto result = std::lower_bound(objs.begin(), objs.end(), id,
45  [](const ref_count_object_heap_ptr &lhs, const service::uid &rhs) {
46  return lhs->id < rhs;

48
