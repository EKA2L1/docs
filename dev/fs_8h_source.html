204  std::shared_ptr<T> get_fs_node_as(const kernel::uid fs_session_handle, const std::uint32_t fs_file_handle) {
205  auto cli_ite = clients.find(fs_session_handle);
206 
207  if (cli_ite == clients.end()) {
208  return nullptr;
209  }
210 
211  auto node = (cli_ite->second)->get_file_node(fs_file_handle);
212  if (node == nullptr) {
213  return nullptr;
214  }
215 
216  return std::reinterpret_pointer_cast<T>(node->vfs_node);
