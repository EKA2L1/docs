73  // Always clear code space with breakpoints, so that if someone accidentally executes
74  // uninitialized, it just breaks into the debugger.
75  void clear_codespace(int offset) {
76  if (is_memory_wx_exclusive()) {
77  change_protection(region, region_size, prot::read_write);
78  }
79 
80  // If not WX Exclusive, no need to call ProtectMemoryPages because we never change the protection from RWX.
81  posion_memory(offset);
82  reset_codeptr(offset);
83

85  // Need to re-protect the part we didn't clear.
86  change_protection(region, offset, prot::read_exec);
87  }
88  }

90  // BeginWrite/EndWrite assume that we keep appending.
91  // If you don't specify a size and we later encounter an executable non-writable block, we're screwed.
92  // These CANNOT be nested. We rely on the memory protection starting at READ|WRITE after start and reset.
93  void begin_write(size_t sizeEstimate = 1) {
94  if (writeStart_) {
95  LOG_ERROR("Can't do nested code block write");
96  }
97 
98  // In case the last block made the current page exec/no-write, let's fix that.
99  if (is_memory_wx_exclusive()) {
100  writeStart_ = get_code_ptr();
101  change_protection(writeStart_, sizeEstimate, prot::read_write);
102  }
103  }
104

106  // OK, we're done. Re-protect the memory we touched.
107  if (PlatformIsWXExclusive() && writeStart_ != nullptr) {
108  const uint8_t *end = get_code_ptr();
109  change_protection(writeStart_, end - writeStart_, prot::read_exec);
110  writeStart_ = nullptr;
111  }
112  }
