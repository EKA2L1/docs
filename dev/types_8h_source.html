155 struct verify_func {
156  template <typename T>
157  bool operator()(T &&value) const {
158  if (std::forward<T>(value)) {

160  }

166 template <uint N>
167 struct verify_impl {
168  const char *cause;
169 
170  template <typename T>
171  auto operator,(T &&value) const {
172  // Verification (can be safely disabled)
173  if (!verify_func()(std::forward<T>(value))) {
174  fmt::raw_verify_error(cause, nullptr, N);
175  }
176 
177  return verify_impl<N + 1>{ cause };
178  }

186 // Verification helper (returns value or lvalue reference, may require to use verify_move instead)
187 template <typename F = verify_func, typename T>
188 inline T verify(const char *cause, T &&value, F &&pred = F()) {
189  if (!pred(std::forward<T>(value))) {
190  using unref = std::remove_const_t<std::remove_reference_t<T>>;
191  fmt::raw_verify_error(cause, fmt::get_type_info<fmt_unveil_t<unref>>(), fmt_unveil<unref>::get(value));
192  }
193 
194  return std::forward<T>(value);
195 }

197 // Verification helper (must be used in return expression or in place of std::move)
