127  void seek(const std::uint64_t amount, seek_where wh) override {
128  if (wh == seek_where::beg) {
129  crr_pos = amount;
130  return;
131  }
132 
133  if (wh == seek_where::cur) {
134  crr_pos += amount;
135  return;
136  }
137 
138  crr_pos = (end - beg) + amount;
139  }
140 
141  std::uint64_t read(void *buf, const std::uint64_t read_size) override {
142  std::uint64_t actual_read_size = common::min(read_size,
143  static_cast<std::uint64_t>(end - beg - crr_pos));
144 
145  memcpy(buf, beg + crr_pos, actual_read_size);
146  crr_pos += actual_read_size;
147 
148  return actual_read_size;
