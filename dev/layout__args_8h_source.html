40  template <typename arg>
41  constexpr std::tuple<arg_layout, layout_args_state> add_to_stack(const layout_args_state &state) {
42  const size_t stack_alignment = alignof(arg);
43  const size_t stack_required = sizeof(arg);
44 
45  const size_t stack_offset = align(state.stack_used, stack_alignment);
46  const size_t next_stack_used = stack_offset + stack_required;
47 
48  const arg_layout layout = { arg_where::stack, stack_offset };
49  const layout_args_state next_state = { state.gpr_used, next_stack_used, state.float_used };

54  template <typename arg>
55  constexpr std::tuple<arg_layout, layout_args_state> add_to_gpr_or_stack(const layout_args_state &state) {
56  const size_t gpr_required = (sizeof(arg) + 3) / 4; // r0 - r3, and stack
57  const size_t gpr_alignment = gpr_required;
58  const size_t gpr_idx = align(state.gpr_used, gpr_alignment);
59  const size_t next_gpr_used = gpr_idx + gpr_required;
60 
61  if (next_gpr_used > 4) {
62  return add_to_stack<arg>(state);
63  }
64 
65  const arg_layout layout = { arg_where::gpr, gpr_idx };
66  const layout_args_state next_state = { next_gpr_used, state.stack_used, state.float_used };
67 
68  return { layout, next_state };
69  }
